        get: function (sprites, spritesheet, prefix, uri) {
            var backgroundImage = uri ? spritesheet : this.settings.filename;

            return templates.less({
                prefix: prefix,
                backgroundImage: backgroundImage,
                sprites: sprites
            });
        }
